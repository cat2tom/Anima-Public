00614                         for (unsigned int j = 0;j < InputImageType::ImageDimension;++j)
00615                             initDir[j] = m_DWIGravityCenter[j] - currentPoint[j];
00616                         break;
00617                 }
00618 
00619                 if (is2d)
00620                     initDir[2] = 0;
00621                 initDir.Normalize();
00622 
00623                 previousDirections[i] = this->InitializeFirstIterationFromModel(initDir,modelValue,numThread);
00624             }
00625 
00626             // Propose a new direction based on the previous one and the diffusion information at current position
00627             double log_prior = 0, log_proposal = 0;
00628             newDirection = this->ProposeNewDirection(previousDirections[i], modelValue, sampling_direction, log_prior,
00629                                                      log_proposal, m_Generators[numThread], numThread);
00630 
00631             // Update the position of the particle
00632             for (unsigned int j = 0;j < InputImageType::ImageDimension;++j)
00633                 currentPoint[j] += m_StepProgression * newDirection[j];
00634 
00635             // Log-weight update must be done at new position (except for prior and proposal)
00636             m_SeedMask->TransformPhysicalPointToContinuousIndex(currentPoint,newIndex);
00637 
00638             // Set the new proposed direction as the current direction
00639             previousDirections[i] = newDirection;
00640 
00641             modelValue.Fill(0.0);
00642 
00643             if (!dwiInterpolators[0]->IsInsideBuffer(newIndex))
00644             {
00645                 fiberComputationData.stoppedParticles[i] = true;
00646                 fiberComputationData.particleWeights[i] = 0;
00647                 continue;
00648             }
00649 
00650             fiberComputationData.fiberParticles[i].push_back(currentPoint);
00651 
00652             estimatedB0Value = this->ComputeModelEstimation(dwiInterpolators, newIndex, dwiValue, estimatedNoiseValue, modelValue);
00653 
00654             // Update the weight of the particle
00655             double updateWeightLogVal = this->ComputeLogWeightUpdate(estimatedB0Value, estimatedNoiseValue, newDirection, sampling_direction,
00656                                                                      modelValue, dwiValue, log_prior, log_proposal, numThread);
00657 
00658             logWeightVals[i] = updateWeightLogVal + anima::safe_log(oldFiberWeights[i]);
00659         }
00660 
00661         // Continue only if some particles are still moving
00662         stopLoop = true;
00663         for (unsigned int i = 0;i < m_NumberOfParticles;++i)
00664         {
00665             if (!fiberComputationData.stoppedParticles[i])
00666             {
00667                 stopLoop = false;
00668                 break;
00669             }
00670         }
00671 
00672         if (stopLoop)
00673             continue;
00674 
00675         emptyClasses.resize(numberOfClasses);
00676         // Computes weight sum for further weight normalization
00677         for (unsigned int i = 0;i < numberOfClasses;++i)
00678         {
00679             emptyClasses[i] = false;
00680             unsigned int classSize = fiberComputationData.reverseClassMemberships[i].size();
00681             tmpVector.clear();
00682 
00683             for (unsigned int j = 0;j < classSize;++j)
00684             {
00685                 if (!fiberComputationData.stoppedParticles[fiberComputationData.reverseClassMemberships[i][j]])
00686                     tmpVector.push_back(logWeightVals[fiberComputationData.reverseClassMemberships[i][j]]);
00687             }
00688 
00689             if (tmpVector.size() != 0)
00690                 logWeightSums[i] = anima::ExponentialSum(tmpVector);
00691             else
00692             {
00693                 logWeightSums[i] = 0;
00694                 emptyClasses[i] = true;
00695             }
00696         }
00697 
00698         // Weight normalization
00699         tmpVector.clear();
00700         for (unsigned int i = 0;i < numberOfClasses;++i)
00701         {
00702             if (!emptyClasses[i])
00703                 tmpVector.push_back(anima::safe_log(fiberComputationData.classWeights[i]) + logWeightSums[i]);
00704         }
00705 
00706         double tmpSum = 0;
00707         double logSumTmpVector = anima::ExponentialSum(tmpVector);
00708 
00709         for (unsigned int i = 0;i < numberOfClasses;++i)
00710         {
00711             if (!emptyClasses[i])
00712             {
00713                 double t = std::exp(anima::safe_log(fiberComputationData.classWeights[i]) + logWeightSums[i] - logSumTmpVector);
00714                 fiberComputationData.classWeights[i] = t;
00715                 tmpSum += t;
00716             }
00717             else
00718                 fiberComputationData.classWeights[i] = 0;
00719         }
00720 
00721         for (unsigned int i = 0;i < numberOfClasses;++i)
00722             fiberComputationData.classWeights[i] /= tmpSum;
00723 
00724         for (unsigned int i = 0;i < m_NumberOfParticles;++i)
00725         {
00726             if (fiberComputationData.stoppedParticles[i])
00727             {
00728                 fiberComputationData.particleWeights[i] = 0;
00729                 continue;
00730             }
00731 
00732             double tmpWeight = logWeightSums[fiberComputationData.classMemberships[i]];
00733 
00734             if (std::isfinite(tmpWeight))
00735                 logWeightVals[i] -= tmpWeight;
00736 
00737             fiberComputationData.particleWeights[i] = std::exp(logWeightVals[i]);
00738             // Q: shouldn't we be treating this case as an empty cluster that shouldn't even exist?
00739         }
00740 
00741         // Resampling if necessary, done class by class
00742         effectiveNumberOfParticles.resize(numberOfClasses);
00743         std::fill(effectiveNumberOfParticles.begin(),effectiveNumberOfParticles.end(),0);
00744         for (unsigned int i = 0;i < m_NumberOfParticles;++i)
00745         {
00746             double weight = fiberComputationData.particleWeights[i];
00747             effectiveNumberOfParticles[fiberComputationData.classMemberships[i]] += weight * weight;
00748         }
00749 
00750         for (unsigned int m = 0;m < numberOfClasses;++m)
00751         {
00752             if (effectiveNumberOfParticles[m] != 0)
00753                 effectiveNumberOfParticles[m] = 1.0 / effectiveNumberOfParticles[m];
00754             else
00755                 continue; // Q: shouldn't we be treating this case as an empty cluster that shouldn't even exist? (same as previous Q)
00756 
00757             // Actual class resampling
00758             if (effectiveNumberOfParticles[m] < m_ResamplingThreshold * fiberComputationData.classSizes[m])
00759             {
00760                 weightSpecificClassValues.resize(fiberComputationData.classSizes[m]);
00761                 previousDirectionsCopy.resize(fiberComputationData.classSizes[m]);
00762                 fiberParticlesCopy.resize(fiberComputationData.classSizes[m]);
00763 
00764                 for (unsigned int i = 0;i < fiberComputationData.classSizes[m];++i)
00765                 {
00766                     unsigned int posIndex = fiberComputationData.reverseClassMemberships[m][i];
00767                     weightSpecificClassValues[i] = fiberComputationData.particleWeights[posIndex];
00768                     previousDirectionsCopy[i] = previousDirections[posIndex];
00769                     fiberParticlesCopy[i] = fiberComputationData.fiberParticles[posIndex];
00770                 }
00771 
00772                 std::discrete_distribution<> dist(weightSpecificClassValues.begin(),weightSpecificClassValues.end());
00773                 usedFibers.resize(fiberComputationData.classSizes[m]);
00774                 std::fill(usedFibers.begin(),usedFibers.end(),false);
00775 
00776                 for (unsigned int i = 0;i < fiberComputationData.classSizes[m];++i)
00777                 {
00778                     unsigned int z = dist(m_Generators[numThread]);
00779                     unsigned int iReal = fiberComputationData.reverseClassMemberships[m][i];
00780                     previousDirections[iReal] = previousDirectionsCopy[z];
00781                     fiberComputationData.fiberParticles[iReal] = fiberParticlesCopy[z];
00782                     // In all of this, we suppose that stopped particles have zero weights and will therefore
00783                     // be lost when resampling
00784                     fiberComputationData.stoppedParticles[iReal] = false;
00785                     usedFibers[z] = true;
00786                 }
00787 
00788                 for (unsigned int i = 0;i < fiberComputationData.classSizes[m];++i)
00789                 {
00790                     unsigned int iReal = fiberComputationData.reverseClassMemberships[m][i];
00791 
00792                     if (!usedFibers[i])
00793                     {
00794                         if (oldFiberWeights[iReal] > m_FiberTrashThreshold / m_NumberOfParticles)
00795                         {
00796                             if (fiberComputationData.particleWeights[iReal] != 0)
00797                                 fiberParticlesCopy[i].pop_back();
00798 
00799                             // The fiber trash used to contain fibers that were lost with a sufficient weight
00800                             // However, using way too much memory so removed for now
00801                             //if (fiberParticlesCopy[i].size() > m_MinLengthFiber / m_StepProgression)
00802                             //    fiberTrash.push_back(fiberParticlesCopy[i]);
00803                         }
00804                     }
00805                 }
00806 
00807                 // Update only weightVals, oldWeightVals will get updated when starting back the loop
00808                 // Same here for stopped fibers, they get rejected when resampling
00809                 for (unsigned int i = 0;i < fiberComputationData.classSizes[m];++i)
00810                     fiberComputationData.particleWeights[fiberComputationData.reverseClassMemberships[m][i]] = 1.0 / fiberComputationData.classSizes[m];
00811             }
00812         }
00813 
00814         // We need stopping criterions
00815         // Length is easy, given that each step is constant we just need to check the fiber size: numIter
00816         // Example :
00817         if (numIter > m_MaxLengthFiber / m_StepProgression)
00818             stopLoop = true;
00819 
00820         numberOfClasses = this->UpdateClassesMemberships(fiberComputationData,previousDirections,m_Generators[numThread]);
00821 
00822         for (unsigned int i = 0;i < fiberComputationData.particleWeights.size();++i)
00823         {
00824             if (!std::isfinite(fiberComputationData.particleWeights[i]))
00825                 itkExceptionMacro("Nan weights after update class membership");
00826         }
00827     }
00828 
00829     // Now that we're done, if we don't keep individual particles, merge them cluster by cluster
00830     if (m_MAPMergeFibers)
00831     {
00832         FiberProcessVectorType mergedOutput,classMergedOutput;
00833         for (unsigned int i = 0;i < numberOfClasses;++i)
00834         {
00835             this->MergeParticleClassFibers(fiberComputationData,classMergedOutput,i);
00836             mergedOutput.insert(mergedOutput.end(),classMergedOutput.begin(),classMergedOutput.end());
00837         }
00838 
00839         fiberComputationData.fiberParticles = mergedOutput;
00840     }
00841 
00842     if (m_MAPMergeFibers)
00843         resultWeights = fiberComputationData.classWeights;
00844     else
00845         resultWeights = fiberComputationData.particleWeights;
00846 
00847     return fiberComputationData.fiberParticles;
00848 }

00850 unsigned int
00851 BaseProbabilisticTractographyImageFilter::UpdateClassesMemberships(FiberWorkType &fiberData, DirectionVectorType &directions, std::mt19937 &random_generator)
00852 {
00853     const unsigned int p = PointType::PointDimension;
00854     typedef anima::KMeansFilter <PointType,p> KMeansFilterType;
00855     unsigned int numClasses = fiberData.classSizes.size();
00856 
00857     // Deciding on cluster merges
00858     FiberProcessVectorType mapMergedFibersRef, mapMergedFibersFlo;
00859     unsigned int newNumClasses = numClasses;
00860 
00861     MembershipType classesFusion(numClasses);
00862     for (unsigned int i = 0;i < numClasses;++i)
00863         classesFusion[i] = i;
00864 
00865     // As described in IPMI, we take the input classes and first try to fuse them
00866     // This is based on a range of possible criterions specified by the user
00867     if (numClasses > 1)
00868     {
00869         for (unsigned int i = 0;i < numClasses;++i)
00870         {
00871             // Fuse test is done on average cluster fiber, if it is an active cluster,
00872             // i.e. at least one of its particle is still moving
00873             bool activeClass = this->MergeParticleClassFibers(fiberData,mapMergedFibersRef,i);
00874             if (!activeClass)
00875                 continue;
00876 
00877             for (unsigned int j = i+1;j < numClasses;++j)
00878             {
00879                 if (classesFusion[j] != j)
00880                     continue;
00881 
00882                 double maxVal = 0;
00883                 bool activeSubClass = this->MergeParticleClassFibers(fiberData,mapMergedFibersFlo,j);
00884                 if (!activeSubClass)
00885                     continue;
00886 
00887                 // Compute a distance between the two clusters, based on user input
00888                 switch (m_ClusterDistance)
00889                 {
00890                     case 0:
00891                     {
00892                         // Former method (quickest)
00893                         unsigned int minSizeFiber = std::min(mapMergedFibersRef[0].size(),mapMergedFibersFlo[0].size());
00894 
00895                         for (unsigned int l = 0;l < minSizeFiber;++l)
00896                         {
00897                             double positionDist = anima::ComputeEuclideanDistance(mapMergedFibersRef[0][l], mapMergedFibersFlo[0][l]);
00898 
00899                             if (positionDist > maxVal)
00900                                 maxVal = positionDist;
00901 
00902                             if (maxVal > m_PositionDistanceFuseThreshold)
00903                                 break;
00904                         }
00905 
00906                         break;
00907                     }
00908 
00909                     case 1:
00910                     {
00911                         // Hausdorff distance
00912                         for (unsigned int l = 0;l < mapMergedFibersRef[0].size();++l)
00913                         {
00914                             double tmpVal = anima::ComputePointToSetDistance(mapMergedFibersRef[0][l], mapMergedFibersFlo[0]);
00915 
00916                             if (tmpVal > maxVal)
00917                                 maxVal = tmpVal;
00918 
00919                             if (maxVal > m_PositionDistanceFuseThreshold)
00920                                 break;
00921                         }
00922 
00923                         if (maxVal <= m_PositionDistanceFuseThreshold)
00924                         {
00925                             for (unsigned int l = 0;l < mapMergedFibersFlo[0].size();++l)
00926                             {
00927                                 double tmpVal = anima::ComputePointToSetDistance(mapMergedFibersFlo[0][l], mapMergedFibersRef[0]);
00928 
00929                                 if (tmpVal > maxVal)
00930                                     maxVal = tmpVal;
00931 
00932                                 if (maxVal > m_PositionDistanceFuseThreshold)
00933                                     break;
00934                             }
00935                         }
00936 
00937                         break;
00938                     }
00939 
00940                     case 2:
00941                     {
00942                         // Modified Hausdorff distance
00943                         maxVal = anima::ComputeModifiedDirectedHausdorffDistance(mapMergedFibersRef[0], mapMergedFibersFlo[0]);
00944 
00945                         if (maxVal <= m_PositionDistanceFuseThreshold)
00946                             maxVal = std::max(maxVal, anima::ComputeModifiedDirectedHausdorffDistance(mapMergedFibersFlo[0], mapMergedFibersRef[0]));
00947 
00948                         break;
00949                     }
00950 
00951                     default:
00952                         break;
00953                 }
00954 
00955                 // If computed distance is smaller than a threshold, we fuse
00956                 // To do so, an index table (classesFusion) is updated, each of its cells tells
00957                 // to which new class the current class belongs. newNumClasses is the new number of classes
00958                 if (maxVal <= m_PositionDistanceFuseThreshold)
00959                 {
00960                     newNumClasses--;
00961                     classesFusion[j] = classesFusion[i];
00962                 }
00963             }
00964         }
00965 
00966         // Some post-processing to have contiguous class numbers as an output
00967         // mapFusion will hold the correspondance between non contiguous and contiguous indexes
00968         int maxVal = -1;
00969         unsigned int currentIndex = 0;
00970         std::map <unsigned int, unsigned int> mapFusion;
00971         for (unsigned int i = 0;i < numClasses;++i)
00972         {
00973             if (maxVal < (int)classesFusion[i])
00974             {
00975                 mapFusion.insert(std::make_pair(classesFusion[i],currentIndex));
00976                 ++currentIndex;
00977                 maxVal = classesFusion[i];
00978             }
00979         }
00980 
00981         for (unsigned int i = 0;i < numClasses;++i)
00982             classesFusion[i] = mapFusion[classesFusion[i]];
00983     }
00984 
00985     std::vector <MembershipType> fusedClassesIndexes(newNumClasses);
00986     for (unsigned int i = 0;i < numClasses;++i)
00987         fusedClassesIndexes[classesFusion[i]].push_back(i);
00988 
00989     // Now we're done with selecting what to fuse
00990     // Therefore, deciding on cluster splits. The trick here is that we don't want to actually really perform fuse
00991     // if it is to split right after (for speed reasons). So we'll play with classesFusion indexes all along
00992     // to keep track of the original indexes.
00993     DirectionVectorType afterMergeClassesDirections(newNumClasses);
00994     MembershipType afterMergeNumPoints(newNumClasses,0);
00995     std::vector <bool> splitClasses(newNumClasses,false);
00996 
00997     Vector3DType zeroDirection(0.0);
00998     std::fill(afterMergeClassesDirections.begin(),afterMergeClassesDirections.end(),zeroDirection);
00999 
01000     // Compute average directions after fusion, directions contains the last directions taken by particles
01001     for (unsigned int i = 0;i < m_NumberOfParticles;++i)
01002     {
01003         if (fiberData.particleWeights[i] == 0)
01004             continue;
01005 
01006         unsigned int classIndex = classesFusion[fiberData.classMemberships[i]];
01007         for (unsigned int j = 0;j < p;++j)
01008             afterMergeClassesDirections[classIndex][j] += directions[i][j];
01009         afterMergeNumPoints[classIndex]++;
01010     }
01011 
01012     unsigned int numSplits = 0;
01013     std::vector < std::pair <unsigned int, double> > afterMergeKappaValues;
01014     // From those averaged directions, we compute a dispersion kappa value, that will be used to decide on split
01015     for (unsigned int i = 0;i < newNumClasses;++i)
01016     {
01017         double norm = 0;
01018         for (unsigned int j = 0;j < p;++j)
01019             norm += afterMergeClassesDirections[i][j] * afterMergeClassesDirections[i][j];
01020         norm = sqrt(norm);
01021 
01022         double R = 1.0;
01023         double kappa = m_KappaSplitThreshold + 1;
01024         if (afterMergeNumPoints[i] != 0)
01025         {
01026             R = norm / afterMergeNumPoints[i];
01027 
01028             if (R*R > 1.0 - 1.0e-16)
01029                 R = sqrt(1.0 - 1.0e-16);
01030 
01031             kappa = std::exp( anima::safe_log(R) + anima::safe_log(p-R*R) - anima::safe_log(1-R*R) );
01032         }
01033 
01034         afterMergeKappaValues.push_back(std::make_pair(i,kappa));
01035         // We do not allow any split resulting in less than m_MinimalNumberOfParticlesPerClass

01042     }

01044     std::partial_sort(afterMergeKappaValues.begin(),afterMergeKappaValues.begin() + numSplits,afterMergeKappaValues.end(),pair_comparator());
01045 
01046     for (unsigned int i = 0;i < numSplits;++i)

01050     // Each split class will be split into two, so new number of classes
01051     // after merge and split is newNumClasses + numSplits

01059 
01060     unsigned int currentIndex = 0;
01061 
01062     FiberType vectorToCluster;
01063     MembershipType clustering;
01064

01066     for (unsigned int i = 0;i < newNumClasses;++i)
01067     {

01069         {
01070             // ith class is just a potential merge of classes. Easy case: just take all particles
01071             // from classes marked as new ith class
01072             for (unsigned int j = 0;j < fusedClassesIndexes[i].size();++j)
01073             {
01074                 unsigned int classIndex = fusedClassesIndexes[i][j];
01075                 for (unsigned int k = 0;k < fiberData.reverseClassMemberships[classIndex].size();++k)
01076                 {
01077                     unsigned int particleNumber = fiberData.reverseClassMemberships[classIndex][k];
01078                     newClassesMemberships[particleNumber] = currentIndex;
01079                     newReverseClassesMemberships[currentIndex].push_back(particleNumber);
01080                 }
01081             }
01082 
01083             if (fusedClassesIndexes[i].size() != 1)
01084             {
01085                 // Recompute class weights after fusion
01086                 newClassWeights[currentIndex] = 0;
01087                 for (unsigned int j = 0;j < fusedClassesIndexes[i].size();++j)
01088                 {
01089                     unsigned int classIndex = fusedClassesIndexes[i][j];
01090                     for (unsigned int k = 0;k < fiberData.reverseClassMemberships[classIndex].size();++k)
01091                         newClassWeights[currentIndex] += fiberData.classWeights[classIndex] * fiberData.particleWeights[fiberData.reverseClassMemberships[classIndex][k]];
01092                 }
01093 
01094                 // Recompute particle weights after fusion
01095                 for (unsigned int j = 0;j < fusedClassesIndexes[i].size();++j)
01096                 {
01097                     unsigned int classIndex = fusedClassesIndexes[i][j];
01098                     for (unsigned int k = 0;k < fiberData.reverseClassMemberships[classIndex].size();++k)
01099                     {
01100                         unsigned int posIndex = fiberData.reverseClassMemberships[classIndex][k];
01101                         newParticleWeights[posIndex] = std::exp(anima::safe_log(fiberData.classWeights[classIndex]) + anima::safe_log(fiberData.particleWeights[posIndex]) - anima::safe_log(newClassWeights[currentIndex]));
01102                     }
01103                 }
01104             }
01105             else
01106                 newClassWeights[currentIndex] = fiberData.classWeights[fusedClassesIndexes[i][0]];
01107

01109         }

01117             for (unsigned int j = 0;j < fusedClassesIndexes[i].size();++j)
01118             {
01119                 unsigned int classIndex = fusedClassesIndexes[i][j];

01121                     vectorToCluster.push_back(fiberData.fiberParticles[fiberData.reverseClassMemberships[classIndex][k]].back());
01122             }

01130             {

01159                 unsigned int classIndex = fusedClassesIndexes[i][j];

01161                 {
01162                     unsigned int classPos = currentIndex + clustering[pos];
01163 
01164                     newClassesMemberships[fiberData.reverseClassMemberships[classIndex][k]] = classPos;
01165                     newReverseClassesMemberships[classPos].push_back(fiberData.reverseClassMemberships[classIndex][k]);
01166 
01167                     newClassWeights[classPos] += fiberData.classWeights[classIndex] * fiberData.particleWeights[fiberData.reverseClassMemberships[classIndex][k]];
01168 
01169                     ++pos;
01170                 }
01171             }
01172
