00073     {
00074         double sumWeights = 0;

00076         if (numWeightsToOptimize == 0)
00077             return 0.0;
00078 
00079         // Strong assumption here that weights are at the beginning of the vector in the parameters
00080         // To do: make this be handled by the multi-compartment model
00081         for (unsigned int i = 0;i < numWeightsToOptimize;++i)
00082         {
00083             sumWeights += dataValue[i];
00084             if (gradValue)
00085                 gradValue[i] = 1.0;
00086         }
00087 
00088         if (gradValue)

00090             for (unsigned int i = numWeightsToOptimize;i < numParameters;++i)
00091                 gradValue[i] = 0;

00093 
00094         // Note: there is always at least one isotropic compartment when estimating weights
00095         return sumWeights - 1.0;
