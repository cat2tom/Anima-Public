00095     double NLOPTOptimizers::NloptFunctionWrapper(unsigned n, const double *x, double *grad, void *data)
00096     {
00097         NLOPTOptimizers *optimizer = static_cast<NLOPTOptimizers *>(data);
00098 
00099         //-----------------------------------------
00100         // Copy the nlopt position to a itk position
00101         // (takes into account the itk scale)
00102         //-----------------------------------------
00103         NLOPTOptimizers::ParametersType itkCurrentPosition(n);
00104         for ( unsigned int i=0; i<n ; i++ )
00105             itkCurrentPosition[i] = x[i]/optimizer->GetScales()[i];
00106 
00107         double f = (optimizer->GetCostFunction()->GetValue(itkCurrentPosition));
00108 
00109         //-----------------------------------------
00110         // If needed, also compute the derivative and copy
00111         // back in *grad
00112         //-----------------------------------------
00113         if ( grad!=NULL )
00114         {
00115             DerivativeType derivative;
00116             optimizer->GetCostFunction()->GetDerivative (itkCurrentPosition, derivative);
00117             for ( unsigned int i=0; i<n ; i++ )
00118                 grad[i] = derivative[i];
00119         }
00120 
00121         return f;
00122     }
